import 'dart:ui';
import 'package:flutter/foundation.dart';
import 'package:flutter/material.dart';

import '../auth/models/course.dart';
import '../auth/models/category.dart' as model;
import '../auth/services/category_service.dart';

class LandingPage extends StatefulWidget {
  const LandingPage({super.key});

  @override
  State<LandingPage> createState() => _LandingPageState();
}

class _LandingPageState extends State<LandingPage> {
  late Future<List<model.Category>> categoriesFuture;
  final TextEditingController _searchController = TextEditingController();
  String _searchQuery = '';

  @override
  void initState() {
    super.initState();
    categoriesFuture = CategoryService.getCategories();
  }

  @override
  void dispose() {
    _searchController.dispose();
    super.dispose();
  }

  void _goToLogin(String role) {
    Navigator.pushNamed(context, "/login", arguments: role);
  }

  void _goToSignupStudent() {
    Navigator.pushNamed(context, "/signup", arguments: "STUDENT");
  }

  void _showEnrollDialog() {
    showDialog(
      context: context,
      builder: (_) => AlertDialog(
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
        title: const Text("Enrollment Required"),
        content: const Text(
          "Please login or signup as a Student to enroll in this course.",
        ),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context),
            child: const Text("Cancel"),
          ),
          ElevatedButton(
            onPressed: () {
              Navigator.pop(context);
              _goToSignupStudent();
            },
            child: const Text("Signup as Student"),
          ),
        ],
      ),
    );
  }

  void _openFooterPage(String key) {
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (_) => FooterDemoPage(pageKey: key),
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    final isMobile = MediaQuery.of(context).size.width < 700;

    return Scaffold(
      body: Container(
        decoration: const BoxDecoration(
          gradient: LinearGradient(
            colors: [Color(0xFF0F2027), Color(0xFF203A43), Color(0xFF2C5364)],
            begin: Alignment.topLeft,
            end: Alignment.bottomRight,
          ),
        ),
        child: CustomScrollView(
          slivers: [
            SliverToBoxAdapter(child: _header()),
            SliverPadding(
              padding: const EdgeInsets.symmetric(horizontal: 24),
              sliver: SliverList(
                delegate: SliverChildListDelegate([
                  const SizedBox(height: 60),
                  _hero(isMobile),
                  const SizedBox(height: 30),
                  _searchBar(),
                  const SizedBox(height: 80),
                  _courseSection(),
                  const SizedBox(height: 100),
                  _footer(),
                ]),
              ),
            ),
          ],
        ),
      ),
    );
  }

  // ================= HEADER =================
  Widget _header() {
    return ClipRRect(
      child: BackdropFilter(
        filter: ImageFilter.blur(sigmaX: 14, sigmaY: 14),
        child: Container(
          padding: const EdgeInsets.symmetric(horizontal: 24, vertical: 16),
          decoration: BoxDecoration(
            color: Colors.white.withOpacity(0.08),
            border: const Border(bottom: BorderSide(color: Colors.white24)),
          ),
          child: Wrap(
            alignment: WrapAlignment.spaceBetween,
            children: [
              Row(
                mainAxisSize: MainAxisSize.min,
                children: const [
                  Icon(Icons.school, color: Colors.white, size: 32),
                  SizedBox(width: 10),
                  Text(
                    "LMS Portal",
                    style: TextStyle(
                      color: Colors.white,
                      fontSize: 24,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                ],
              ),
              Wrap(
                spacing: 10,
                children: [
                  _roleButton("Student", Icons.person, "STUDENT"),
                  _roleButton("Instructor", Icons.school, "INSTRUCTOR"),
                  _roleButton("Admin", Icons.admin_panel_settings, "ADMIN"),
                ],
              ),
            ],
          ),
        ),
      ),
    );
  }

  // ================= HERO =================
  Widget _hero(bool isMobile) {
    return Column(
      children: [
        Text(
          "Upgrade Your Skills.\nTeach the World.",
          textAlign: TextAlign.center,
          style: TextStyle(
            fontSize: isMobile ? 30 : 44,
            fontWeight: FontWeight.bold,
            color: Colors.white,
          ),
        ),
        const SizedBox(height: 20),
        Text(
          "A modern Learning Management System for everyone",
          textAlign: TextAlign.center,
          style: TextStyle(color: Colors.grey.shade300),
        ),
        const SizedBox(height: 36),
        Wrap(
          spacing: 16,
          children: [
            _ctaButton(
              text: "Login",
              filled: true,
              onTap: () => Navigator.pushNamed(context, "/login"),
            ),
            _ctaButton(
              text: "Create Free Account",
              filled: false,
              onTap: _goToSignupStudent,
            ),
          ],
        ),
      ],
    );
  }

  // ================= SEARCH =================
  Widget _searchBar() {
    return TextField(
      controller: _searchController,
      onChanged: (v) => setState(() => _searchQuery = v.toLowerCase()),
      style: const TextStyle(color: Colors.white),
      decoration: InputDecoration(
        hintText: "Search courses...",
        hintStyle: const TextStyle(color: Colors.white70),
        prefixIcon: const Icon(Icons.search, color: Colors.white),
        filled: true,
        fillColor: Colors.white.withOpacity(0.12),
        border: OutlineInputBorder(
          borderRadius: BorderRadius.circular(16),
          borderSide: BorderSide.none,
        ),
      ),
    );
  }

  // ================= COURSES =================
  Widget _courseSection() {
    return FutureBuilder<List<model.Category>>(
      future: categoriesFuture,
      builder: (_, snapshot) {
        if (!snapshot.hasData) {
          return const Center(child: CircularProgressIndicator());
        }

        return Column(
          children: snapshot.data!.map((category) {
            final courses = category.courses.where((c) {
              return _searchQuery.isEmpty ||
                  c.title.toLowerCase().contains(_searchQuery);
            }).toList();

            if (courses.isEmpty) return const SizedBox();

            return Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(category.name,
                    style: const TextStyle(
                        color: Colors.white,
                        fontSize: 26,
                        fontWeight: FontWeight.bold)),
                const SizedBox(height: 10),
                GridView.builder(
                  shrinkWrap: true,
                  physics: const NeverScrollableScrollPhysics(),
                  gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
                    crossAxisCount:
                    MediaQuery.of(context).size.width > 900 ? 3 : 1,
                    crossAxisSpacing: 24,
                    mainAxisSpacing: 24,
                    childAspectRatio: 1.2,
                  ),
                  itemCount: courses.length,
                  itemBuilder: (_, i) => CourseCardPremium(
                    course: courses[i],
                    onEnrollTap: _showEnrollDialog,
                  ),
                ),
                const SizedBox(height: 50),
              ],
            );
          }).toList(),
        );
      },
    );
  }

  // ================= FOOTER =================
  Widget _footer() {
    return Container(
      padding: const EdgeInsets.symmetric(vertical: 60, horizontal: 24),
      decoration: BoxDecoration(
        gradient: const LinearGradient(
          colors: [Color(0xFF283593), Color(0xFF1E3E63)],
          begin: Alignment.topLeft,
          end: Alignment.bottomRight,
        ),
        borderRadius: BorderRadius.circular(24),
      ),
      child: Column(
        children: [
          Wrap(
            spacing: 60,
            runSpacing: 20,
            alignment: WrapAlignment.center,
            children: [
              _footerColumn("Platform", ["Courses", "Instructors", "Pricing"]),
              _footerColumn("Company", ["About", "Careers", "Support"]),
              _footerColumn("Legal", ["Privacy Policy", "Terms"]),
            ],
          ),
          const SizedBox(height: 30),
          const Text(
            "© 2026 LMS Portal • Learn • Teach • Manage",
            style: TextStyle(color: Colors.white70, fontSize: 12),
          ),
        ],
      ),
    );
  }

  Widget _footerColumn(String title, List<String> items) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Text(title,
            style: const TextStyle(
                color: Colors.white,
                fontWeight: FontWeight.bold,
                fontSize: 16)),
        const SizedBox(height: 12),
        ...items.map(
              (e) => GestureDetector(
            onTap: () => _openFooterPage(e),
            child: Padding(
              padding: const EdgeInsets.only(bottom: 8),
              child: Text(
                e,
                style: const TextStyle(
                  color: Colors.white70,
                  decoration: TextDecoration.underline,
                ),
              ),
            ),
          ),
        ),
      ],
    );
  }

  Widget _roleButton(String label, IconData icon, String role) {
    return TextButton.icon(
      onPressed: () => _goToLogin(role),
      icon: Icon(icon, color: Colors.white),
      label: Text(label, style: const TextStyle(color: Colors.white)),
    );
  }

  Widget _ctaButton({
    required String text,
    required bool filled,
    required VoidCallback onTap,
  }) {
    return SizedBox(
      width: 220,
      height: 48,
      child: filled
          ? ElevatedButton(
          onPressed: onTap,
          style: ElevatedButton.styleFrom(
              backgroundColor: Colors.blueAccent,
              shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(14))),
          child: Text(text,
              style: const TextStyle(fontWeight: FontWeight.bold)))
          : OutlinedButton(
          onPressed: onTap,
          style: OutlinedButton.styleFrom(
              side: const BorderSide(color: Colors.blueAccent, width: 2),
              shape: RoundedRectangleBorder(
                  borderRadius: BorderRadius.circular(14))),
          child: Text(text,
              style: const TextStyle(fontWeight: FontWeight.bold))),
    );
  }
}

/* ======================================================
   PREMIUM COURSE CARD
====================================================== */

class CourseCardPremium extends StatefulWidget {
  final Course course;
  final VoidCallback onEnrollTap;

  const CourseCardPremium({
    super.key,
    required this.course,
    required this.onEnrollTap,
  });

  @override
  State<CourseCardPremium> createState() => _CourseCardPremiumState();
}

class _CourseCardPremiumState extends State<CourseCardPremium> {
  bool _hovered = false;

  @override
  Widget build(BuildContext context) {
    final isFree = widget.course.price == 0;

    return MouseRegion(
      onEnter: (_) => setState(() => _hovered = true),
      onExit: (_) => setState(() => _hovered = false),
      child: AnimatedContainer(
        duration: const Duration(milliseconds: 200),
        transform: _hovered && kIsWeb
            ? (Matrix4.identity()..translate(0, -10))
            : Matrix4.identity(),
        decoration: BoxDecoration(
          gradient: const LinearGradient(
              colors: [Color(0xFF1565C0), Color(0xFF283593)]),
          borderRadius: BorderRadius.circular(22),
          boxShadow: [
            BoxShadow(
                color: Colors.black45,
                offset: const Offset(0, 12),
                blurRadius: _hovered ? 30 : 18),
          ],
        ),
        child: Padding(
          padding: const EdgeInsets.all(20),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Align(
                alignment: Alignment.topRight,
                child: Chip(
                  backgroundColor: isFree ? Colors.green : Colors.orange,
                  label: Text(isFree ? "FREE" : "PAID",
                      style: const TextStyle(color: Colors.white)),
                ),
              ),
              const Spacer(),
              Text(widget.course.title,
                  maxLines: 2,
                  overflow: TextOverflow.ellipsis,
                  style: const TextStyle(
                      color: Colors.white,
                      fontSize: 20,
                      fontWeight: FontWeight.bold)),
              const SizedBox(height: 6),
              const Text("Instructor: LMS Expert",
                  style: TextStyle(color: Colors.white70)),
              const SizedBox(height: 6),
              Row(
                children: List.generate(
                    5,
                        (_) =>
                    const Icon(Icons.star, color: Colors.amber, size: 18)),
              ),
              const Spacer(),
              Row(
                mainAxisAlignment: MainAxisAlignment.spaceBetween,
                children: [
                  Text(isFree ? "FREE" : "₹ ${widget.course.price}",
                      style: const TextStyle(
                          color: Colors.white,
                          fontSize: 18,
                          fontWeight: FontWeight.bold)),
                  ElevatedButton(
                      onPressed: widget.onEnrollTap,
                      child: const Text("Enroll")),
                ],
              ),
            ],
          ),
        ),
      ),
    );
  }
}
