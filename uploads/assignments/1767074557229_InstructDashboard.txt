// ---------------------- fetchAll student-------------
async function fetchAllStudents() {
  try {
    // api already injects Authorization via interceptor, so just call GET with default config
    const res = await api.get("/instructor/students");
    setAllStudents(res.data || []);
  } catch (err) {
    console.error("Failed to fetch students", err);
    setAllStudents([]);
  }
}

// ---------------- FETCH ALL DISCUSSIONS ----------------
async function fetchDiscussions(courseId) {
  if (!courseId) return;

  try {
    const res = await api.get(
      `/instructor/course/${courseId}/discussions`
    );
    setDiscussions(Array.isArray(res.data) ? res.data : []);
  } catch (err) {
    console.error("Fetch discussions failed", err);
    setDiscussions([]);
  }
}

 async function fetchCourseProgress(courseId) {
  try {

    console.error("Failed to fetch progress", err);
    setCourseProgress({ progress: 0 });
  }
}


  // ---------------- FETCH LESSONS ----------------
  async function fetchLessons(moduleId) {
    if (!moduleId) return;
    try {
      const res = await api.get(`/instructor/module/${moduleId}/lessons`);
      setCourseDetails((prev) => {
        const updated = { ...prev };
        const module = updated[selectedCourse.id]?.find((m) => String(m.id) === String(moduleId));
        if (module) module.lessons = res.data || [];
        return updated;
      });
    } catch (err) {
      console.warn("fetchLessons error", err);
    }
  }

async function handleCreateDiscussion(e) {
  e.preventDefault();

  try {
    await fetchAllStudents();
    const payload = {
      question: discussionPayload.question,
      course: { id: discussionPayload.courseId || selectedCourse?.id },
      student: discussionPayload.studentId ? { id: discussionPayload.studentId } : null,
    };
    if (!payload.question || !payload.course.id) {
      return alert("Please enter a question and select a course.");
    }

    await api.post(`/instructor/discussion/create`, payload);
    setShowDiscussionModal(false);
    setDiscussionPayload({ question: "", courseId: null, studentId: null });
    if (selectedCourse?.id) await fetchDiscussions(selectedCourse.id);

    alert("Discussion created.");
  } catch (err) {
    console.error("Failed to create discussion", err);
    const msg = err?.response?.data?.message || "Failed to create discussion (see console)";
    alert(msg);
  }
}

        <ModalWrap onClose={() => setShowQuizModal(false)}>
          <h5>Add Quiz (lesson selected)</h5>
          <form onSubmit={handleAddQuiz}>
            <input className="form-control mb-2" placeholder="Question" required value={quizPayload.question}
              onChange={(e) => setQuizPayload((s) => ({ ...s, question: e.target.value }))} />
            <input className="form-control mb-2" placeholder="Option A" required value={quizPayload.optionA}
              onChange={(e) => setQuizPayload((s) => ({ ...s, optionA: e.target.value }))} />
            <input className="form-control mb-2" placeholder="Option B" required value={quizPayload.optionB}
              onChange={(e) => setQuizPayload((s) => ({ ...s, optionB: e.target.value }))} />
            <input className="form-control mb-2" placeholder="Option C" value={quizPayload.optionC}
              onChange={(e) => setQuizPayload((s) => ({ ...s, optionC: e.target.value }))} />
            <input className="form-control mb-2" placeholder="Option D" value={quizPayload.optionD}
              onChange={(e) => setQuizPayload((s) => ({ ...s, optionD: e.target.value }))} />
            <select className="form-select mb-2" value={quizPayload.correctAnswer}
              onChange={(e) => setQuizPayload((s) => ({ ...s, correctAnswer: e.target.value }))}>
              <option value="A">A</option>
              <option value="B">B</option>
              <option value="C">C</option>
              <option value="D">D</option>
            </select>
            <div className="text-end">
              <button type="button" className="btn btn-secondary me-2" onClick={() => setShowQuizModal(false)}>Cancel</button>
              <button className="btn btn-primary">Add Quiz</button>
            </div>
          </form>
        </ModalWrap>
      )}

      
      {showAssignmentModal && (
        <ModalWrap onClose={() => setShowAssignmentModal(false)}>
          <h5>Add Assignment (lesson selected)</h5>
          <form onSubmit={handleAddAssignment}>
            <input className="form-control mb-2" placeholder="Title" required value={assignmentPayload.title}
              onChange={(e) => setAssignmentPayload((s) => ({ ...s, title: e.target.value }))} />
            <textarea className="form-control mb-2" placeholder="Description" required value={assignmentPayload.description}
              onChange={(e) => setAssignmentPayload((s) => ({ ...s, description: e.target.value }))} />
            <input type="date" className="form-control mb-2" value={assignmentPayload.dueDate}
              onChange={(e) => setAssignmentPayload((s) => ({ ...s, dueDate: e.target.value }))} />
            <div className="text-end">
              <button type="button" className="btn btn-secondary me-2" onClick={() => setShowAssignmentModal(false)}>Cancel</button>
              <button className="btn btn-primary">Add Assignment</button>
            </div>
          </form>
        </ModalWrap>
      )}

      
    {showEnrollModal && (
  <ModalWrap onClose={() => setShowEnrollModal(false)}>
    <h5 className="mb-3">🎓 Enroll Student</h5>

    <form onSubmit={handleEnrollStudent}>
      {/* Student */}
      <label className="form-label fw-semibold">Student</label>
      <select
        className="form-select mb-3"
        required
        value={enrollPayload.student.id}
        onChange={(e) =>
          setEnrollPayload((s) => ({
            ...s,
            student: { id: Number(e.target.value) },
          }))
        }
      >
        <option value="">-- Select Student --</option>

        {allStudents.length === 0 ? (
          <option disabled>No students found</option>
        ) : (
          allStudents.map((stu) => (
            <option key={stu.id} value={stu.id}>
              {stu.name} ({stu.email})
            </option>
          ))
        )}
      </select>

      {/* Amount Paid */}
      <label className="form-label fw-semibold">Amount Paid (₹)</label>
      <input
        type="number"
        className="form-control mb-3"
        min="0"
        required
        value={enrollPayload.amountPaid}
        onChange={(e) =>
          setEnrollPayload((s) => ({
            ...s,
            amountPaid: Number(e.target.value),
          }))
        }
      />

      {/* Progress */}
      <label className="form-label fw-semibold">Initial Progress (%)</label>
      <input
        type="number"
        className="form-control mb-3"
        min="0"
        max="100"
        value={enrollPayload.progressPercentage}
        onChange={(e) =>
          setEnrollPayload((s) => ({
            ...s,
            progressPercentage: Number(e.target.value),
          }))
        }
      />

      {/* Actions */}
      <div className="text-end">
        <button
          type="button"
          className="btn btn-secondary me-2"
          onClick={() => setShowEnrollModal(false)}
        >
          Cancel
        </button>

        <button type="submit" className="btn btn-primary">
          Enroll Student
        </button>
      </div>
    </form>
  </ModalWrap>
)}

      
  {showDiscussionModal && (
  <ModalWrap onClose={() => setShowDiscussionModal(false)}>
    <h5>Create Discussion</h5>

    <textarea
      className="form-control mb-3"
      rows={4}
      placeholder="Enter discussion question"
      value={discussionPayload.question}
      onChange={(e) =>
        setDiscussionPayload((s) => ({ ...s, question: e.target.value }))
      }
    />

    {/* Optional: Select Course */}
    {!selectedCourse && (
      <select
        className="form-select mb-3"
        value={discussionPayload.courseId || ""}
        onChange={(e) =>
          setDiscussionPayload((s) => ({
            ...s,
            courseId: e.target.value,
          }))
        }
      >
        <option value="">Select Course</option>
        {courses.map((c) => (
          <option key={c.id} value={c.id}>
            {c.title}
          </option>
        ))}
      </select>
    )}

     {/* 🔍 SEARCH STUDENT */}
    <input
      className="form-control mb-2"
      placeholder="Search student by name"
      value={studentSearch}
      onChange={(e) => setStudentSearch(e.target.value.toLowerCase())}
    />

    {/* 👤 SELECT STUDENT */}
    <select
      className="form-select mb-3"
      value={discussionPayload.studentId || ""}
      onChange={(e) =>
        setDiscussionPayload((s) => ({
          ...s,
          studentId: e.target.value || null,
        }))
      }
    >
      <option value="">All Students</option>

      {/* 👇 ADD IT EXACTLY HERE */}
      {allStudents
        .filter((s) =>
          s.name.toLowerCase().includes(studentSearch)
        )
        .map((s) => (
          <option key={s.id} value={s.id}>
            {s.name} ({s.email})
          </option>
        ))}
    </select>

    <div className="text-end">
      <button
        className="btn btn-secondary me-2"
        onClick={() => setShowDiscussionModal(false)}
      >
        Cancel
      </button>

      <button
        className="btn btn-primary"
        onClick={handleCreateDiscussion}
      >
        Post
      </button>
    </div>
  </ModalWrap>
)}


   {showNotificationModal && (
  <ModalWrap onClose={() => setShowNotificationModal(false)}>
    <h5>Send Notification</h5>

    <input
      className="form-control mb-2"
      placeholder="Title"
      value={notificationPayload.title}
      onChange={(e) =>
        setNotificationPayload((s) => ({ ...s, title: e.target.value }))
      }
    />

    <textarea
      className="form-control mb-2"
      placeholder="Message"
      value={notificationPayload.message}
      onChange={(e) =>
        setNotificationPayload((s) => ({ ...s, message: e.target.value }))
      }
    />

    <select
      className="form-select mb-2"
      value={notificationPayload.sendTo}
      onChange={(e) =>
        setNotificationPayload((s) => ({ ...s, sendTo: e.target.value }))
      }
    >
      <option value="ALL">All Users</option>
      <option value="ROLE">Specific Role</option>
    </select>

    {notificationPayload.sendTo === "ROLE" && (
      <select
        className="form-select mb-2"
        value={notificationPayload.role}
        onChange={(e) =>
          setNotificationPayload((s) => ({ ...s, role: e.target.value }))
        }
      >
        <option value="">Select Role</option>
        <option value="STUDENT">Student</option>
        <option value="INSTRUCTOR">Instructor</option>
      </select>
    )}

    <div className="text-end">
      <button
        className="btn btn-secondary me-2"
        onClick={() => setShowNotificationModal(false)}
      >
        Cancel
      </button>

      <button
        className="btn btn-primary"
        onClick={
          notificationPayload.sendTo === "ALL"
            ? sendNotificationToAll
            : sendNotificationToRole
        }
      >
        Send
      </button>
    </div>
  </ModalWrap>
)}



      
      {showReplyModal && (
        <ModalWrap onClose={() => setShowReplyModal(false)}>
          <h5>Reply to Discussion</h5>
          <form onSubmit={handleReplyDiscussion}>
            <input className="form-control mb-2" placeholder="Discussion ID" required value={selectedDiscussionId || ""}
              onChange={(e) => setSelectedDiscussionId(e.target.value)} />
            <textarea className="form-control mb-2" placeholder="Message" required value={replyPayload.message}
              onChange={(e) => setReplyPayload((s) => ({ ...s, message: e.target.value }))} />
            <div className="text-end">
              <button type="button" className="btn btn-secondary me-2" onClick={() => setShowReplyModal(false)}>Cancel</button>
              <button className="btn btn-primary">Reply</button>
            </div>
          </form>
        </ModalWrap>
      )}
    </div>
  );
}

// Modal helper
function ModalWrap({ children, onClose }) {
  return (
    <div className="modal-backdrop-custom">
      <div className="modal-custom">
        <div className="d-flex justify-content-between align-items-center mb-3">
          <div />
          <button className="btn-close" onClick={onClose} />
        </div>
        {children}
      </div>
    </div>
  );
}

function modulesForSelectedCourse_static(detail, selectedCourse) {
  if (!selectedCourse || !detail) return [];
  return detail?.modules || detail?.course?.modules || detail?.moduleList || [];
}
